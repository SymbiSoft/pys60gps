Directory size function:

import os
def getDirectorySize(directory):
 dir_size = 0
 for (path, dirs, files) in os.walk(directory):
  for file in files:
   filename = os.path.join(path, file)
   dir_size += os.path.getsize(filename)
 return dir_size
